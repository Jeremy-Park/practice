Alert
    dont use its bad
    use console instead
    console.log('hello world');
    console.error('this is an error');
    console.warn('this is a warning');

variables
    dont use var
    let and const have block level scope
    let can be reassigned
    let age = 39;

    use const unless you want to reassign

destructuring
    const person = {
        firstName: 'john',
        lastName: 'doe',
        address: {
            street: 'finch',
            city: 'toronto'
        }
    }

    const {firstName, lastName, address: {city}} = person;
    console.log(city);
